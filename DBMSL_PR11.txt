mapReduce in monododb:

1. Sum of all salaries per city
Ans:
var mapFunction = function(){
	emit(this.emp_city, this.emp_salary);
};

var reduceFunction = function(key, value){
	return Array.sum(value);
};

db.Employee.mapReduce(mapFunction, reduceFunction, {out: "salary_by_city"});
db.salary_by_city.find().pretty();

2. Max salary per city
Ans:
var mapFunction = function(){
	emit(this.emp_city, this.emp_salary);
};

var reduceFunction = function(key, value){
