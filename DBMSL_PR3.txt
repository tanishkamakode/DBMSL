STUDENT:
-- 1. Student details and Placement details using NATURAL JOIN
SELECT * 
FROM Student NATURAL JOIN PlacementDrive;

-- 2. All student details with company name who have conducted in same drive
SELECT s.*, p.Pcompany_name
FROM Student s
JOIN PlacementDrive p ON s.Drive_id = p.Drive_id;

-- 3. Student name and branch having package 5 LPA
SELECT s.s_name, s.s_branch
FROM Student s
JOIN PlacementDrive p ON s.Drive_id = p.Drive_id
WHERE p.Package = 500000;

-- 4. Student names and company names with training fee > 20000
SELECT s.s_name, t.Tcompany_name
FROM Student s
JOIN Training t ON s.T_id = t.T_id
WHERE t.TFee > 20000;

-- 5. All training details attended by "Shantanu" in 2011
SELECT t.*
FROM Student s
JOIN Training t ON s.T_id = t.T_id
WHERE s.s_name = 'Shantanu' AND YEAR(t.T_date) = 2011;

-- 6. Total number of companies who conducted training before 2015
SELECT COUNT(DISTINCT Tcompany_name) AS TotalCompanies
FROM Training
WHERE YEAR(T_date) < 2015;

-- 7. Students with company 'Microsoft' and location 'Thane'
SELECT s.s_name
FROM Student s
JOIN PlacementDrive p ON s.Drive_id = p.Drive_id
WHERE p.Pcompany_name = 'Microsoft' AND p.Location = 'Thane';

-- 8. Names of all students who joined 'Microsoft' training in 2015
SELECT s.s_name
FROM Student s
JOIN Training t ON s.T_id = t.T_id
WHERE t.Tcompany_name = 'Microsoft' AND YEAR(t.T_date) = 2015;

-- 9. Create view showing Student and Training details
CREATE VIEW StudentTrainingView AS
SELECT s.s_id, s.s_name, s.s_branch, t.Tcompany_name, t.TFee, t.T_date
FROM Student s
JOIN Training t ON s.T_id = t.T_id;
